검색 코드 예시

이 페이지는 실제로 box-db.js 의 300개 DB를 필터링해 결과를 표시합니다. 클릭만 되는 모형이 아니라, 입력값을 바탕으로 검색과 추천이 동작합니다.

const results = SD_BOX_DB.filter(box =>
  (!category || box.category === category) &&
  (!shape || box.shape === shape) &&
  (!flute || box.flute === flute) &&
  (!use || box.use === use) &&
  (!industry || box.industry === industry) &&
  (!qty || box.minQty <= qty)
)
