현재 사이트에 적용된 검색엔진과 추천 알고리즘 v2의 핵심 로직을 정리한 페이지입니다.

완전한 박스 검색엔진 코드

function filterBoxes(input){
  return SD_BOX_DB.filter(box =>
    (!input.category || box.category === input.category) &&
    (!input.shape || box.shape === input.shape) &&
    (!input.flute || box.flute === input.flute) &&
    (!input.use || box.use === input.use) &&
    (!input.industry || box.industry === input.industry) &&
    (!input.qty || box.minQty <= input.qty) &&
    (!input.w || box.width >= input.w) &&
    (!input.l || box.length >= input.l) &&
    (!input.h || box.height >= input.h)
  );
}

추천 알고리즘 v2

function scoreBoxV2(box, input){
  let score = 0;
  if(input.w && box.width >= input.w) score += Math.max(0, 100 - (box.width-input.w)*0.7);
  if(input.l && box.length >= input.l) score += Math.max(0, 100 - (box.length-input.l)*0.7);
  if(input.h && box.height >= input.h) score += Math.max(0, 80 - (box.height-input.h)*0.9);
  if(input.category && box.category === input.category) score += 90;
  if(input.shape && box.shape === input.shape) score += 70;
  if(input.flute && box.flute === input.flute) score += 60;
  if(input.use && box.use === input.use) score += 55;
  if(input.industry && box.industry === input.industry) score += 45;
  if(input.qty && box.minQty <= input.qty) score += 40; else if(input.qty) score -= 20;
  if(input.recommendedFlute && box.flute === input.recommendedFlute) score += 35;
  return score;
}
